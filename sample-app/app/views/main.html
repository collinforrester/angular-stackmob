Make Sure You:

1. Set your public API key

StackmobProvider.setApiKey('YOUR_API_KEY');

2. Set your environment

StackmobProvider.setEnvironment('0');

3. Login


loggedInUser = {{loggedInUser | json}} 

$scope.login = function() {
    Stackmob.login($scope.user.username, $scope.user.password)
        .then(function(data) {
            $scope.loggedInUser = data.stackmob.user;
            $scope.user = {};
        });
};

4. Fire away!

            
$scope.queryPersons = function() {
    // remember, Person is a glorified $resource object!
    var Person = Stackmob.schema('person');
    $scope.people = Person.query();
};
